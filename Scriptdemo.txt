-- Load UI Library (Ví dụ sử dụng Fluent hoặc Rayfield)
local Fluent = loadstring(game:HttpGet("https://github.com/dawid-scripts/Fluent/releases/latest/download/main.lua"))()

local Window = Fluent:CreateWindow({
    Title = "My Custom Hub",
    SubTitle = "by YourName",
    TabWidth = 160,
    Size = UDim2.fromOffset(580, 460),
    Acrylic = true,
    Theme = "Dark"
})

-- Tạo một Tab mới
local Tabs = {
    Main = Window:AddTab({ Title = "Main", Icon = "home" }),
    Settings = Window:AddTab({ Title = "Settings", Icon = "settings" })
}

-- Thêm tính năng Auto Farm (Ví dụ đơn giản)
local AutoFarmToggle = Tabs.Main:AddToggle("AutoFarm", {Title = "Auto Farm Level", Default = false })

AutoFarmToggle:OnChanged(function()
    _G.AutoFarm = Fluent.Options.AutoFarm.Value
    
    spawn(function()
        while _G.AutoFarm do
            wait()
            -- Logic tìm quái và đánh quái sẽ nằm ở đây
            print("Đang tự động farm...")
        end
    end)
end)
